### T2 合成贤者之石

二分+贪心

```
import java.util.Scanner;
// 1:无需package
// 2: 类名必须Main, 不可修改

public class Main {
    static boolean check(int cnt,int s){
      return s >= cnt*3+3;
    }
    static int ans = 0;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        for(int i=0;i<n;i++){
          int x = sc.nextInt();
          int l = 0,r = x;
          while(l<r){
            int mid = l+r+1>>1;
            if (check(mid,x))
              l = mid;
            else r = mid-1;
          }
          ans += l;
        }
        
      System.out.println(ans);
    }
}
```

### T3 Yaya与加减法

贪心+细节	

第一个数不选、加减号个数是会超的

```
import java.util.*;
// 1:无需package
// 2: 类名必须Main, 不可修改


public class Main {
  static int N = 101000;
static int[] arr = new int[N];
static long ans = 0;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int a = sc.nextInt();
        int b = sc.nextInt();
        for(int i=1;i<=n;i++){
          arr[i] = sc.nextInt();
        }
        Arrays.sort(arr,2,n+1);
        ans = arr[1];
        int i;
        for(i=2;i<=n;i++){
          if(arr[i]>0 || b==0) break;
          if(b>0){
            ans += -arr[i];
            b -= 1;
          }
        }
        int j;
        for(j=n;j>=i;j--){
          if(arr[i]<0 ||a==0) break;
          if(a>0){
            ans += arr[j];
            a -= 1;
          }
        }
        if(a==0){
          for(int k=i;k<=j;k++)
            ans += -arr[k];
        }else if(b==0){
          for(int k=i;k<=j;k++)
            ans += arr[k];
        }
        System.out.println(ans);
    }
}
```



### T4 Yaya与字符画

二分答案+爆搜+记忆化搜索

二维地图、、V字形、

```
import java.util.Scanner;
// 1:无需package
// 2: 类名必须Main, 不可修改

public class Main {
    static int N = 2020;
    static char[][] g = new char[N][N];
    static int[][][] f = new int[N][N][2];
    static int n,m;
    static int[] dx = {-1,-1};
    static int[] dy = {-1,1};

    //(x,y)向着一个方向pos的最大长度(包括当前节点)
    static int dfs(int x,int y,int pos){
      if(f[x][y][pos]!= -1) return f[x][y][pos];
      int res = 1;
      int nx = x+dx[pos];
      int ny = y+dy[pos];
      if(nx>=1 && nx<=n && ny>=1 && ny<=m && g[nx][ny]=='*'){
        res += dfs(nx,ny,pos);
      }

      return f[x][y][pos] = res;
    }

    static boolean check(int sz){
      for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
          if (g[i][j]!='*') continue;
          int sz1 = dfs(i,j,0);
          int sz2 = dfs(i,j,1);
          if(Math.min(sz1,sz2)-1>=sz) return true;
        }
      }
      return false;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        m = sc.nextInt();
        for(int i=1;i<=n;i++){
          char[] t = sc.next().toCharArray();
          for(int j=1;j<=m;j++)
            g[i][j] = t[j-1];
        }
        for(int i=0;i<N;i++)
          for(int j=0;j<N;j++)
            f[i][j][0] = f[i][j][1] = -1;
        int l=0,r =4000;
        while(l<r){
          int mid = l+r+1>>1;
          if (check(mid)) l = mid;
          else r = mid-1;
        }
        System.out.println(l);
    }
}
```



### T5 咒语融合（WA）

https://www.lanqiao.cn/problems/18444/learning/?contest_id=183

题意：找出两个数，使得这两个数的和 减去 （两数相同位之和）最大

暴力+剪枝+贪心

贪心：这两个数一定都很大，最多减5*9=45，所以如果结果比ans-45还小，那么就直接跳出

```
import java.util.*;
// 1:无需package
// 2: 类名必须Main, 不可修改

public class Main {
  static int[] arr = new int[101000];
  static long ans;
  static long solve(int a,int b){
    long res = a+b;
    while(a>0 && b>0){
      if(a%10 == b%10)
        res -= a%10;
        a/=10;
        b/=10;
    }
    return res;
  }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        for(int i=0;i<n;i++)
          arr[i] = sc.nextInt();
        Arrays.sort(arr,0,n);
      for(int i=n-1;i>=0;i--){
        for(int j=i-1;j>=0;j--){
          long val = solve(arr[i],arr[j]);
          if(val > ans)
            ans = val;
          if(val < ans-50)
            break;
        }
      }
      System.out.println(ans);
    }
}
```
